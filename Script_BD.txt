CREATE DATABASE MUSIC_BOT;

USE MUSIC_BOT; 

CREATE TABLE LOG (
ID INT PRIMARY KEY auto_increment NOT NULL,
CODIGO INT NOT NULL,
BOT VARCHAR(60) NOT NULL,
NOME VARCHAR(60) NOT NULL,
AUTOR VARCHAR(60) NOT NULL,
DATE DATE NOT NULL,
TIME TIME NOT NULL,
DESCRICAO VARCHAR (120)
);

CREATE TABLE COMANDO (
CODIGO INT PRIMARY KEY auto_increment,
NOME VARCHAR(60)
);

CREATE TABLE MUSICA(
ID INT PRIMARY KEY auto_increment NOT NULL,
CODIGO INT NOT NULL,
BOT VARCHAR(60) NOT NULL,
CANAL_DE_VOZ(60) NOT NULL,
NOME VARCHAR(180) NOT NULL,
CANAL VARCHAR(60) NOT NULL,
DURACAO VARCHAR (12) NOT NULL,
VISUALIZACAO VARCHAR (12) NOT NULL,
LINK VARCHAR (120),
DATE DATE NOT NULL,
TIME TIME NOT NULL
);

ALTER TABLE `LOG` 
ADD CONSTRAINT `fk_codigo` FOREIGN KEY ( `CODIGO` ) 
REFERENCES `COMANDO` ( `CODIGO` ) ;
